\simpleslide{What/who is ``fuzzy''?}{%
\clearbox{1,13}{\includegraphics[width=7\semcm]{fuzzy.jpg}}
\textbox{9,13}{15}{\small%
{\bf fuzz$\bullet$y} (fuz´e), {\em adj.} {\bf fuzz$\bullet$i$\bullet$er},
{\bf fuzz$\bullet$i$\bullet$est}. {\bf 1.} of the nature of or resembling
fuzz: A soft, fuzzy material. {\bf 2.} covered with fuzz: a plant with
broad, fuzzy leaves. {\bf 3.} indistinct; blurred: A fuzzy photograph
usually means you jiggled the camera. {\bf 4.} muddleheaded; incoherent,
often as a result of intoxication: a fuzzy thinker; to become fuzzy
after one drink.}}
\end{slide}
